

-- EXERCÍCIO
CREATE TABLE treinos (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    user_id TEXT NOT NULL,
    data DATE NOT NULL,
    categoria TEXT NOT NULL,
    qualidade TEXT NOT NULL,
    calorias_gastas FLOAT,
    observacoes TEXT,
    criado_em TIMESTAMPTZ DEFAULT now()
);
CREATE INDEX idx_treinos_user_id ON treinos(user_id);

CREATE TABLE peso_semanal (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    user_id TEXT NOT NULL,
    semana_inicio DATE NOT NULL,
    peso_kg FLOAT NOT NULL,
    criado_em TIMESTAMPTZ DEFAULT now()
);
CREATE INDEX idx_peso_user_id ON peso_semanal(user_id);

-- ALIMENTAÇÃO
CREATE TABLE refeicoes_diarias (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    user_id TEXT NOT NULL,
    data DATE NOT NULL,
    desvio_plano BOOLEAN DEFAULT FALSE NOT NULL,
    observacoes_desvio TEXT,
    criado_em TIMESTAMPTZ DEFAULT now()
);
CREATE INDEX idx_refeicoes_user_id ON refeicoes_diarias(user_id);

CREATE TABLE itens_refeicao (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    refeicao_diaria_id INTEGER REFERENCES refeicoes_diarias(id),
    tipo_refeicao TEXT NOT NULL,
    alimento TEXT NOT NULL,
    quantidade_g FLOAT NOT NULL,
    calorias FLOAT NOT NULL,
    proteinas_g FLOAT,
    carboidratos_g FLOAT,
    gorduras_g FLOAT,
    criado_em TIMESTAMPTZ DEFAULT now()
);

-- FINANÇAS
CREATE TABLE transacoes (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    user_id TEXT NOT NULL,
    data DATE NOT NULL,
    tipo TEXT NOT NULL,
    valor FLOAT NOT NULL,
    categoria TEXT NOT NULL,
    descricao TEXT,
    criado_em TIMESTAMPTZ DEFAULT now()
);
CREATE INDEX idx_transacoes_user_id ON transacoes(user_id);

CREATE TABLE centros_custo (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    user_id TEXT NOT NULL,
    categoria TEXT NOT NULL,
    limite_mensal FLOAT NOT NULL,
    criado_em TIMESTAMPTZ DEFAULT now()
);
CREATE INDEX idx_centros_user_id ON centros_custo(user_id);

-- Configuração financeira mensal por usuário
CREATE TABLE configuracao_financeira (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    user_id TEXT NOT NULL,
    renda_mensal FLOAT NOT NULL,
    mes_referencia DATE NOT NULL,  -- sempre o primeiro dia do mês (ex: 2026-03-01)
    criado_em TIMESTAMPTZ DEFAULT now()
);
CREATE INDEX idx_config_financeira_user_id ON configuracao_financeira(user_id);
